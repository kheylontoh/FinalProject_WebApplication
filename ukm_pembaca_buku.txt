function renderBukuTable() {

  const isAdmin = currentUser.role === 'admin';
  const rows = document.getElementById('buku-tbody');

  let filtered = books.filter(b => {
    const matchQ = !q || b.judul.toLowerCase().includes(q) || b.penulis.toLowerCase().includes(q);
    const matchG = !g || b.genre === g;
    return matchQ && matchG;
  });

  if (filtered.length === 0) {
    rows.innerHTML = `<tr><td colspan="6" style="text-align:center;color:var(--ink-light);padding:2rem">Tidak ada buku ditemukan.</td></tr>`;
    return;

  rows.innerHTML = filtered.map((b, i) => `
    <tr>
      <td style="color:var(--ink-light)">${i+1}</td>
      <td>
        <div class="book-title">${b.judul}</div>

      <td><span class="genre-tag">${b.genre}</span></td>

function renderLoans() {
  const isAdmin = currentUser.role === 'admin';
  const myLoans = isAdmin ? loans : loans.filter(l => l.userId === currentUser.id);
  const container = document.getElementById('my-loans');

  if (myLoans.length === 0) {
    container.innerHTML = `<div class="empty-state"><div class="ico">📋</div><p>Belum ada riwayat peminjaman.</p></div>`;
    return;
  }

  const statusLabel = { active:'Aktif', overdue:'Terlambat', pending:'Menunggu', done:'Selesai' };
  const statusClass = { active:'lb-active', overdue:'lb-overdue', pending:'lb-pending', done:'lb-done' };
  const dotClass = { active:'dot-active', overdue:'dot-overdue', pending:'dot-pending', done:'dot-done' };

  container.innerHTML = myLoans.map(l => {
    const book = books.find(b => b.id === l.bookId);
    const user = USERS.find(u => u.id === l.userId);
    const bookName = book ? book.judul : '(Buku dihapus)';
    const userInfo = isAdmin ? `<span style="color:var(--amber)">· ${user ? user.name : '?'}</span> ` : '';
    return `
    <div class="loan-item">
      <div class="loan-dot ${dotClass[l.status]}"></div>
      <div class="loan-info">
        <div class="title">${bookName}</div>
        <div class="meta">${userInfo}Pinjam: ${l.tanggalPinjam} · Tenggat: ${l.tenggat}</div>

            <button class="btn btn-amber btn-sm" onclick="approveLoans(${l.id})">✔ Setujui</button>
            <button class="btn btn-danger btn-sm" onclick="rejectLoan(${l.id})">✖ Tolak</button>
          </div>` : ''}
        ${!isAdmin && l.status === 'active' ? `
          <button class="btn btn-outline btn-sm" style="margin-top:0.5rem" onclick="openKembali(${l.id})">↩ Kembalikan</button>` : ''}
      </div>
      <span class="loan-badge ${statusClass[l.status]}">${statusLabel[l.status]}</span>
    </div>`;
  }).join('');
}

function submitPinjam() {
  const bookId = parseInt(document.getElementById('pinjam-book').value);
  const durasi = parseInt(document.getElementById('pinjam-durasi').value);
  const catatan = document.getElementById('pinjam-catatan').value.trim();
  if (!bookId) { showAlert('alert-pinjam', 'Pilih buku terlebih dahulu.', 'error'); return; }

  const already = loans.find(l => l.userId === currentUser.id && l.bookId === bookId && (l.status === 'active' || l.status === 'pending'));
  if (already) { showAlert('alert-pinjam', 'Anda sudah meminjam buku ini.', 'error'); return; }

  const today = new Date();
  const tenggat = new Date(today.getTime() + durasi * 86400000);
  const fmt = d => d.toISOString().split('T')[0];

  loans.push({ id: loanIdCounter++, userId: currentUser.id, bookId, tanggalPinjam: fmt(today), tenggat: fmt(tenggat), status: 'pending', catatan });
  document.getElementById('pinjam-book').value = '';
  document.getElementById('pinjam-catatan').value = '';
  renderLoans();
  showAlert('alert-pinjam', 'Permintaan peminjaman berhasil diajukan! Tunggu konfirmasi admin.', 'success');
}

function approveLoans(id) {
  const l = loans.find(x => x.id === id);
  l.status = 'active';
  const book = books.find(b => b.id === l.bookId);
  if (book) book.stok = Math.max(0, book.stok - 1);
  renderLoans();
  renderStats();
  renderBukuTable();
  showAlert('alert-pinjam', 'Peminjaman disetujui.', 'success');
}

function rejectLoan(id) {
  loans = loans.filter(l => l.id !== id);
  renderLoans();
  showAlert('alert-pinjam', 'Permintaan peminjaman ditolak.', 'error');
}

function openKembali(id) {
  returningLoanId = id;
  const l = loans.find(x => x.id === id);
  const b = books.find(x => x.id === l.bookId);
  document.getElementById('kembali-title').textContent = b ? b.judul : '(Buku)';
  openModal('modal-kembali');
}

function konfirmasiKembali() {
  const l = loans.find(x => x.id === returningLoanId);
  l.status = 'done';
  const b = books.find(x => x.id === l.bookId);
  if (b) b.stok++;
  closeModal('modal-kembali');
  renderLoans();
  renderStats();
  renderBukuTable();
  renderBerandaBooks();
  showAlert('alert-pinjam', 'Buku berhasil dikembalikan. Terima kasih!', 'success');
}

function goPage(p) {
  document.querySelectorAll('.page').forEach(el => el.classList.remove('active'));
  document.querySelectorAll('.nav-link').forEach(el => el.classList.remove('active'));
  document.getElementById('page-' + p).classList.add('active');

  const navMap = { beranda:0, buku:1, peminjaman:2, tim:3 };
  document.querySelectorAll('.nav-link')[navMap[p]].classList.add('active');
}

function openModal(id) { document.getElementById(id).classList.add('open'); }
function closeModal(id) { document.getElementById(id).classList.remove('open'); }

function showAlert(containerId, msg, type) {
  const el = document.getElementById(containerId);
  el.innerHTML = `<div class="alert alert-${type === 'success' ? 'success' : 'error'}">${msg}</div>`;
  setTimeout(() => { el.innerHTML = ''; }, 4000);
}

document.querySelectorAll('.modal-overlay').forEach(overlay => {

  });
});
